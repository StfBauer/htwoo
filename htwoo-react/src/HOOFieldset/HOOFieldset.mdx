# HOOFieldset

A HOOFieldset component will create a HOOLegend component and then render all child elements below it.

## Arguments

<ArgTypes of={HOOFieldset} />

export const Template = (args) => <HOOFieldset {...args}>
<HOOField>
  <HOOLabel label='My Field A' />
  <HOOInputDesc description='My field A description' />
  <HOOText value="Temp"/>
  <HOOValidationMsg validationMsg='My Validation Message'/>
</HOOField>
<HOOField>
  <HOOLabel label='My Field B' />
  <HOOInputDesc description='My field B description' />
  <HOOText value="Temp"/>
  <HOOValidationMsg validationMsg='My Validation Message'/>
</HOOField>
</HOOFieldset>;
