# HOOIconOverflow

## Flyout Menu Items

Enabling flyout menu depends on passing an instance of `HOOFlyoutMenu` as a child

## Arguments

<ArgTypes of={HOOIconOverflow} />

export const OverflowTemplate = (args) => <HOOIconOverflow {...args}>
    <HOOFlyoutMenu contextItems={[{ iconName: 'hoo-icon-ninjacat', label: 'First Element' }, { iconName: 'hoo-icon-ninjacat', label: 'Second Element' }]} 
    contextItemClicked={(ev, contextItem) => { alert(JSON.stringify(contextItem)) }}/>
  </HOOIconOverflow>;
