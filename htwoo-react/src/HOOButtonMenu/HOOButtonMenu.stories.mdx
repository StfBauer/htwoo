# HOOButtonMenu

## Flyout Menu Items

This depicts using `HOOButtonMenu`.

## Arguments

<ArgTypes of={HOOButtonMenu} />

export const OverflowTemplate = (args) => <HOOButtonMenu {...args} />;

## Basic
<Canvas>
  <Story 
    name="Basic"
    args = {
      (function () {
        symset.initSymbols();
        const dropdownItems = [
          { iconName: 'hoo-icon-ninjacat', label: 'First Element' }, 
          { iconName: 'hoo-icon-ninjacat', label: 'Second Element' }
        ];
        const options = dropdownItems;

        return {
          contextItems: options,
          contextItemClicked: (event, menuItem) => { console.log(`Flyout menu item: ${JSON.stringify(menuItem)}`) }
        };
      })()
    }>
    {OverflowTemplate.bind({})}
  </Story>
</Canvas>
